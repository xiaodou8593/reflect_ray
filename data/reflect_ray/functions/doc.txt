# 激光的数据模板
{
	particle:"dust 1 0.1 0.1 0.25",
	rate:100,
	duration:100,
	step:0.01d
}

particle: string
	控制激光的粒子

rate: int
	控制递归多少次

step: double
	控制递归每步前进多少格方块

duration: int
	控制激光存在多少tick
